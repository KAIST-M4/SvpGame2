    <script>
        const badWords = [
            "관종이냐", "진짜 극혐이다", "낄끼빠빠 못하네", "눈치 좀 챙겨라", 
            "왜 사냐 진짜", "아무도 너 안 좋아해", "나대지 마", "진짜 노답이다",
            "찌질해", "얼굴 실화냐", "꼴도 보기 싫어", "다 너 욕하더라",
            "생각 좀 하고 살아", "수준 떨어져", "제발 좀 꺼져"
        ];

        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(el => el.classList.add('hidden'));
            document.getElementById(screenId).classList.remove('hidden');
        }

        /* =========================================================
           테트리스 로직
        ========================================================= */
        const canvas = document.getElementById('tetris');
        const context = canvas.getContext('2d');
        context.scale(20, 20); 

        const bgContainer = document.getElementById('bg-words-container');
        let linesClearedTotal = 0;
        let dropCounter = 0;
        let dropInterval = 1000;
        let lastTime = 0;
        let tetrisFrameId;
        let isTetrisPlaying = false;

        function createPiece(type) {
            if (type === 'T') return [[0, 0, 0], [1, 1, 1], [0, 1, 0]];
            if (type === 'O') return [[2, 2], [2, 2]];
            if (type === 'L') return [[0, 3, 0], [0, 3, 0], [0, 3, 3]];
            if (type === 'J') return [[0, 4, 0], [0, 4, 0], [4, 4, 0]];
            if (type === 'I') return [[0, 5, 0, 0], [0, 5, 0, 0], [0, 5, 0, 0], [0, 5, 0, 0]];
            if (type === 'S') return [[0, 6, 6], [6, 6, 0], [0, 0, 0]];
            if (type === 'Z') return [[7, 7, 0], [0, 7, 7], [0, 0, 0]];
        }
        const colors = [null, '#a855f7', '#eab308', '#f97316', '#3b82f6', '#06b6d4', '#22c55e', '#ef4444'];

        function createMatrix(w, h) {
            const matrix = [];
            while (h--) matrix.push(new Array(w).fill(0));
            return matrix;
        }
        const arena = createMatrix(12, 20);

        const player = { pos: {x: 0, y: 0}, matrix: null };

        function drawMatrix(matrix, offset) {
            matrix.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value !== 0) {
                        context.fillStyle = colors[value];
                        context.fillRect(x + offset.x, y + offset.y, 1, 1);
                        context.strokeStyle = 'rgba(0,0,0,0.5)';
                        context.lineWidth = 0.05;
                        context.strokeRect(x + offset.x, y + offset.y, 1, 1);
                    }
                });
            });
        }

        function drawTetris() {
            context.fillStyle = '#0f172a';
            context.fillRect(0, 0, canvas.width, canvas.height);
            drawMatrix(arena, {x: 0, y: 0});
            
            // --- 잔상(Ghost) 계산 및 그리기 시작 ---
            // 1. 현재 플레이어의 위치와 모양을 복사한 가상의 블록(ghost) 생성
            const ghost = {
                matrix: player.matrix,
                pos: { x: player.pos.x, y: player.pos.y }
            };
            
            // 2. 충돌하기 전까지 가상 블록을 계속 아래로 내림
            while (!collide(arena, ghost)) {
                ghost.pos.y++;
            }
            ghost.pos.y--; // 충돌 직전의 위치로 한 칸 되돌림

            // 3. 투명도를 조절하여 잔상 그리기
            context.globalAlpha = 0.25; // 투명도 25% 설정
            drawMatrix(ghost.matrix, ghost.pos);
            context.globalAlpha = 1.0;  // 투명도 원래대로(100%) 복구
            // --- 잔상(Ghost) 그리기 끝 ---

            // 실제 플레이어 블록 그리기
            drawMatrix(player.matrix, player.pos);
        }

        function merge(arena, player) {
            player.matrix.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value !== 0) {
                        arena[y + player.pos.y][x + player.pos.x] = value;
                    }
                });
            });
        }

        function collide(arena, player) {
            const [m, o] = [player.matrix, player.pos];
            for (let y = 0; y < m.length; ++y) {
                for (let x = 0; x < m[y].length; ++x) {
                    if (m[y][x] !== 0 && (arena[y + o.y] && arena[y + o.y][x + o.x]) !== 0) {
                        return true;
                    }
                }
            }
            return false;
        }

        function playerDrop() {
            if(!isTetrisPlaying) return;
            player.pos.y++;
            if (collide(arena, player)) {
                player.pos.y--;
                merge(arena, player);
                playerReset();
                arenaSweep();
            }
            dropCounter = 0;
        }

        function playerHardDrop() {
            if(!isTetrisPlaying) return;
            while (!collide(arena, player)) { player.pos.y++; }
            player.pos.y--;
            merge(arena, player);
            playerReset();
            arenaSweep();
            dropCounter = 0;
        }

        function playerMove(dir) {
            if(!isTetrisPlaying) return;
            player.pos.x += dir;
            if (collide(arena, player)) player.pos.x -= dir;
        }

        function playerRotate() {
            if(!isTetrisPlaying) return;
            const pos = player.pos.x;
            let offset = 1;
            rotate(player.matrix);
            while (collide(arena, player)) {
                player.pos.x += offset;
                offset = -(offset + (offset > 0 ? 1 : -1));
                if (offset > player.matrix[0].length) {
                    rotate(player.matrix, -1);
                    player.pos.x = pos;
                    return;
                }
            }
        }

        function rotate(matrix, dir = 1) {
            for (let y = 0; y < matrix.length; ++y) {
                for (let x = 0; x < y; ++x) {
                    [matrix[x][y], matrix[y][x]] = [matrix[y][x], matrix[x][y]];
                }
            }
            if (dir > 0) matrix.forEach(row => row.reverse());
            else matrix.reverse();
        }

        function playerReset() {
            const pieces = 'ILJOTSZ';
            player.matrix = createPiece(pieces[pieces.length * Math.random() | 0]);
            player.pos.y = 0;
            player.pos.x = (arena[0].length / 2 | 0) - (player.matrix[0].length / 2 | 0);
            
            if (collide(arena, player)) {
                isTetrisPlaying = false;
                cancelAnimationFrame(tetrisFrameId);
                canvas.style.opacity = "0.5";
                setTimeout(() => showScreen('end-screen'), 1000);
            }
        }

        function arenaSweep() {
            let linesThisTurn = 0;
            outer: for (let y = arena.length - 1; y >= 0; --y) {
                for (let x = 0; x < arena[y].length; ++x) {
                    if (arena[y][x] === 0) continue outer;
                }
                const row = arena.splice(y, 1)[0].fill(0);
                arena.unshift(row);
                ++y;
                linesThisTurn++;
            }

            if (linesThisTurn > 0) {
                linesClearedTotal += linesThisTurn;
                document.getElementById('lines-cleared').innerText = linesClearedTotal;
                if(dropInterval > 200) dropInterval -= 20; 

                for(let i=0; i<linesThisTurn; i++) {
                    spawnPermanentBackgroundWord();
                }
            }
        }

        function spawnPermanentBackgroundWord() {
            const wordEl = document.createElement('div');
            wordEl.className = 'bg-word';
            wordEl.innerText = badWords[Math.floor(Math.random() * badWords.length)];
            
            wordEl.style.left = Math.random() * 80 + '%';
            wordEl.style.top = Math.random() * 80 + '%';
            wordEl.style.fontSize = (Math.random() * 2 + 1.5) + 'em'; // 모바일 고려 글자 크기 약간 축소
            wordEl.style.transform = `rotate(${Math.random() * 40 - 20}deg)`;
            
            bgContainer.appendChild(wordEl);
        }

        function updateTetris(time = 0) {
            if (!isTetrisPlaying) return;
            const deltaTime = time - lastTime;
            lastTime = time;
            dropCounter += deltaTime;
            if (dropCounter > dropInterval) playerDrop();
            
            drawTetris();
            tetrisFrameId = requestAnimationFrame(updateTetris);
        }

        function startTetrisGame() {
            showScreen('tetris-screen');
            arena.forEach(row => row.fill(0));
            linesClearedTotal = 0;
            dropInterval = 1000;
            document.getElementById('lines-cleared').innerText = '0';
            canvas.style.opacity = "1";
            bgContainer.innerHTML = ''; 
            
            playerReset();
            isTetrisPlaying = true;
            updateTetris();
        }

        /* =========================================================
           터치 & 스와이프 조작 (모바일)
        ========================================================= */
        let touchStartX = 0;
        let touchStartY = 0;
        const swipeThreshold = 30; // 스와이프로 인식할 최소 거리

        document.getElementById('tetris-container').addEventListener('touchstart', function(e) {
            if(!isTetrisPlaying) return;
            // 버튼 터치는 무시 (버튼 이벤트가 별도로 처리됨)
            if(e.target.tagName === 'BUTTON') return; 
            
            touchStartX = e.changedTouches[0].screenX;
            touchStartY = e.changedTouches[0].screenY;
            e.preventDefault(); // 화면 스크롤 방지
        }, {passive: false});

        document.getElementById('tetris-container').addEventListener('touchend', function(e) {
            if(!isTetrisPlaying || e.target.tagName === 'BUTTON') return;
            
            let touchEndX = e.changedTouches[0].screenX;
            let touchEndY = e.changedTouches[0].screenY;
            
            let deltaX = touchEndX - touchStartX;
            let deltaY = touchEndY - touchStartY;

            // X축 이동이 더 클 경우 (좌우 스와이프)
            if (Math.abs(deltaX) > Math.abs(deltaY)) {
                if (Math.abs(deltaX) > swipeThreshold) {
                    if (deltaX > 0) playerMove(1); // 오른쪽
                    else playerMove(-1);           // 왼쪽
                } else {
                    // 제자리 짧은 터치는 회전으로 간주
                     playerRotate();
                }
            } 
            // Y축 이동이 더 클 경우 (상하 스와이프)
            else {
                if (Math.abs(deltaY) > swipeThreshold) {
                    if (deltaY > 0) playerDrop(); // 아래로 살짝 밀면 한 칸 내리기
                    else playerRotate();          // 위로 밀면 회전
                } else {
                     // 제자리 짧은 터치는 회전
                     playerRotate();
                }
            }
        }, {passive: false});


        // PC 키보드 조작
        document.addEventListener('keydown', event => {
            if (!isTetrisPlaying) return;
            if(["Space", "ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight"].includes(event.code)) {
                event.preventDefault(); 
            }
            if (event.keyCode === 37) playerMove(-1);
            else if (event.keyCode === 39) playerMove(1);
            else if (event.keyCode === 40) playerDrop();
            else if (event.keyCode === 38) playerRotate();
            else if (event.keyCode === 32) playerHardDrop();
        });

        function submitReflection() {
            const reflection = document.getElementById('reflection-input').value;
            if(reflection.trim() === "") {
                alert("당신의 솔직한 감정을 한 줄이라도 꼭 적어주세요.");
                return;
            }
            showScreen('thanks-screen');
        }
    </script>
